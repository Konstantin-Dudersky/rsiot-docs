22/// Базовое устройство для опроса по шине.
23pub struct DeviceBase<TMsg, TFieldbusRequest, TFieldbusResponse, TBuffer>
24where
25    TMsg: MsgDataBound,
26    TFieldbusRequest: RequestResponseBound,
27{
28    /// Запросы при инициализации устройства
29    pub fn_init_requests: fn(&TBuffer) -> Vec<TFieldbusRequest>,
30
31    /// Периодические запросы
32    pub periodic_requests: Vec<ConfigPeriodicRequest<TFieldbusRequest, TBuffer>>,
33
34    /// Обновление буфера на основе входящих сообщений
35    ///
36    /// Обычно соответствует параметру fn_input конфигурации устройства
37    ///
38    /// # Пример
39    ///
40    /// ```rust
41    /// let msgs = [Custom::AllInputs(buffer.all_inputs)];
42    /// let msgs = msgs.iter().map(|m| Message::new_custom(*m)).collect();
43    /// msgs
44    /// ```
45    pub fn_msgs_to_buffer: fn(&TMsg, &mut TBuffer),
46
47    /// Периодическое формирование запросов на основе fn_buffer_to_request
48    pub buffer_to_request_period: Duration,
49
50    /// Преобразование данных из буфера в массив запросов на шине
51    ///
52    /// Вызывается несколькими способами:
53    ///
54    /// - при обновлении буфера на основе входящих сообщений функцией fn_msgs_to_buffer
55    ///
56    /// - при расшифровке ответа от устройства, при возвращении true из функции fn_response_to_buffer
57    ///
58    /// - периодически с периодом buffer_to_request_period
59    pub fn_buffer_to_request: fn(&TBuffer) -> anyhow::Result<Vec<TFieldbusRequest>>,
60
61    /// Обновление буфера на основе данных, полученных с устройства
62    pub fn_response_to_buffer:
63        fn(TFieldbusResponse, &mut TBuffer) -> anyhow::Result<ResponseResult>,
64
65    /// Функция создания сообщений на основе буфера
66    ///
67    /// Обычно соответствует параметру fn_output кофигурации устройства
68    ///
69    /// Пример:
70    ///
71    /// ```rust
72    /// |buffer| {vec![]}
73    /// ```
74    pub fn_buffer_to_msgs: fn(&mut TBuffer) -> Vec<TMsg>,
75
76    /// Вывод информации о состоянии устройства
77    pub device_state_output: Option<ConfigDeviceStateOutput<TMsg>>,
78
79    /// Значения в буфере при инициализации
80    pub buffer_default: TBuffer,
81}

83impl<TMsg, TRequest, TResponse, TBuffer> DeviceBase<TMsg, TRequest, TResponse, TBuffer>
84where
85    TRequest: 'static + RequestResponseBound,
86    TResponse: 'static + RequestResponseBound,
87    TMsg: MsgDataBound + 'static,
88    TBuffer: 'static + BufferBound,
89{
90    /// Запустить работу
91    pub async fn spawn(
92        self,
93        id: impl AsRef<str>,
94        ch_rx_msgbus_to_device: MsgBusInput<TMsg>,
95        ch_tx_device_to_fieldbus: mpsc::Sender<TRequest>,
96        ch_rx_fieldbus_to_device: mpsc::Receiver<TResponse>,
97        ch_tx_device_to_msgbus: mpsc::Sender<Message<TMsg>>,
98    ) -> super::Result<()> {
99        let buffer = self.buffer_default;
100        let buffer = Arc::new(Mutex::new(buffer));
101
102        let device_state = DeviceState {
103            init_completed: false,
104            response_ok_count: 0,
105            response_err_count: 0,
106            avg_request_duration: Duration::default(),
107        };
108        let device_state = Arc::new(Mutex::new(device_state));
109
110        let (ch_tx_need_request, ch_rx_need_request) = mpsc::channel::<()>(100);
111        let (ch_tx_request, ch_rx_request) = mpsc::channel::<TRequest>(100);
112        let (ch_tx_output_to_filter, ch_rx_output_to_filter) = mpsc::channel::<Message<TMsg>>(500);
113
114        let mut task_set: JoinSet<super::Result<()>> = JoinSet::new();
115
116        // Задача выполнения первоначальных запросов
117        //
118        // Приостанавливаем выполнение, пока не будет выполнена задача
119        let task = tasks::InitRequest {
120            id: id.as_ref().to_string(),
121            buffer: buffer.clone(),
122            device_state: device_state.clone(),
123            fn_init_requests: self.fn_init_requests,
124            ch_tx_request: ch_tx_request.clone(),
125        };
126        join_set_spawn(
127            &mut task_set,
128            format!("master_device | init_request | {}", id.as_ref()),
129            task.spawn(),
130        );
131
132        // Задача создания запросов на основе входящих сообщений
133        let task = tasks::InputRequest {
134            buffer: buffer.clone(),
135            device_state: device_state.clone(),
136            ch_rx_msgbus_to_device,
137            ch_tx_need_request: ch_tx_need_request.clone(),
138            fn_msgs_to_buffer: self.fn_msgs_to_buffer,
139        };
140        join_set_spawn(
141            &mut task_set,
142            format!("master_device | input_request | {}", id.as_ref()),
143            task.spawn(),
144        );
145
146        // Задача периодического формирования запросов на основе буфера
147        let task = tasks::BufferPeriodic {
148            device_state: device_state.clone(),
149            ch_tx_need_request: ch_tx_need_request.clone(),
150            period: self.buffer_to_request_period,
151        };
152        join_set_spawn(
153            &mut task_set,
154            format!("master_device | buffer_periodic | {}", id.as_ref()),
155            task.spawn(),
156        );
157
158        // Задача создания периодических запросов
159        for periodic_request in self.periodic_requests {
160            let task = tasks::PeriodicRequest {
161                device_state: device_state.clone(),
162                buffer: buffer.clone(),
163                period: periodic_request.period,
164                fn_request: periodic_request.fn_requests,
165                ch_tx_request: ch_tx_request.clone(),
166            };
167            join_set_spawn(
168                &mut task_set,
169                format!("master_device | periodic_request | {}", id.as_ref()),
170                task.spawn(),
171            );
172        }
173
174        // Задача формирования запросов на основе буфера
175        let task = tasks::BufferToRequests {
176            buffer: buffer.clone(),
177            ch_rx_buffer_changed: ch_rx_need_request,
178            ch_tx_request: ch_tx_request.clone(),
179            fn_buffer_to_request: self.fn_buffer_to_request,
180        };
181        join_set_spawn(
182            &mut task_set,
183            format!("master_device | buffer_to_requests | {}", id.as_ref()),
184            task.spawn(),
185        );
186
187        // Задача отправки запросов
188        let task = tasks::Request {
189            ch_rx_request,
190            ch_tx_device_to_fieldbus,
191        };
192        join_set_spawn(
193            &mut task_set,
194            format!("master_device | request | {}", id.as_ref()),
195            task.spawn(),
196        );
197
198        // Задача обработки ответа
199        let task = tasks::Response {
200            buffer: buffer.clone(),
201            device_state: device_state.clone(),
202            ch_rx_fieldbus_to_device,
203            ch_tx_output_to_filter: ch_tx_output_to_filter.clone(),
204            ch_tx_need_request: ch_tx_need_request.clone(),
205            fn_response_to_buffer: self.fn_response_to_buffer,
206            fn_buffer_to_msgs: self.fn_buffer_to_msgs,
207        };
208        join_set_spawn(
209            &mut task_set,
210            format!("master_device | response | {}", id.as_ref()),
211            task.spawn(),
212        );
213
214        // Вывод состояния устройства
215        if let Some(cdso) = self.device_state_output {
216            let task = tasks::DeviceStateOutput {
217                device_state: device_state.clone(),
218                ch_tx_output_to_filter: ch_tx_output_to_filter.clone(),
219                config: cdso,
220            };
221            join_set_spawn(
222                &mut task_set,
223                format!("master_device | device_state_output | {}", id.as_ref()),
224                task.spawn(),
225            );
226        }
227
228        // Задачи фильтрации одинаковых сообщений
229        let task = shared_tasks::filter_identical_data::FilterIdenticalData {
230            input: ch_rx_output_to_filter,
231            output: ch_tx_device_to_msgbus,
232        };
233        join_set_spawn(
234            &mut task_set,
235            format!("master_device | filter_identical_data | {}", id.as_ref()),
236            task.spawn().map_err(super::Error::TaskFilterIdenticalData),
237        );
238
239        while let Some(res) = task_set.join_next().await {
240            res??;
241        }
242        Ok(())
243    }
244}

246impl<TMsg, TFieldbusRequest, TFieldbusResponse, TBuffer> Default
247    for DeviceBase<TMsg, TFieldbusRequest, TFieldbusResponse, TBuffer>
248where
249    TFieldbusRequest: RequestResponseBound,
250    TBuffer: Default,
251    TMsg: MsgDataBound,
252{
253    fn default() -> Self {
254        DeviceBase {
255            fn_init_requests: |_| vec![],
256            periodic_requests: vec![],
257            fn_msgs_to_buffer: |_, _| (),
258            buffer_to_request_period: Duration::from_millis(1000),
259            fn_buffer_to_request: |_| Ok(vec![]),
260            fn_response_to_buffer: |_, _| ResponseResult::ok(),
261            fn_buffer_to_msgs: |_| vec![],
262            device_state_output: None,
263            buffer_default: Default::default(),
264        }
265    }
