1use std::sync::{atomic::AtomicU8, Arc};
2
3use futures::TryFutureExt;
4use tokio::{sync::mpsc, task::JoinSet};
5
6use crate::{
7    components::shared_tasks,
8    executor::{join_set_spawn, CmpInOut},
9    message::MsgDataBound,
10};
11
12use super::{tasks, Config};
13
14const BUFFER_SIZE: usize = 100;
15
16pub async fn fn_process<TMsg>(config: Config<TMsg>, msg_bus: CmpInOut<TMsg>) -> super::Result<()>
17where
18    TMsg: 'static + MsgDataBound,
19{
20    let mut task_set: JoinSet<super::Result<()>> = JoinSet::new();
21
22    let partner_live_counter = Arc::new(AtomicU8::new(0));
23
24    let (ch_tx_msgbus_to_input, ch_rx_msgbus_to_input) = mpsc::channel(BUFFER_SIZE);
25    let (ch_tx_to_msgbus, ch_rx_to_msgbus) = mpsc::channel(BUFFER_SIZE);
26
27    // Передаем входящие сообщения в канал mpsc
28    let task = shared_tasks::msgbus_to_mpsc::MsgBusToMpsc {
29        msg_bus: msg_bus.clone(),
30        output: ch_tx_msgbus_to_input,
31    };
32    join_set_spawn(
33        &mut task_set,
34        task.spawn().map_err(super::Error::TaskMsgBusToMpsc),
35    );
36
37    // Обновляем счетчик на основе входящих сообщений
38    let task = tasks::FindPartnerCounter {
39        input: ch_rx_msgbus_to_input,
40        fn_find_partner_counter: config.fn_find_partner_counter,
41        live_counter: partner_live_counter.clone(),
42    };
43    join_set_spawn(&mut task_set, task.spawn());
44
45    // Периодическая проверка счетчика
46    let task = tasks::CheckPartnerPeriod {
47        output: ch_tx_to_msgbus.clone(),
48        fn_check_partner_counter: config.fn_check_partner_counter,
49        check_partner_period: config.check_partner_period,
50        live_counter: partner_live_counter,
51    };
52    join_set_spawn(&mut task_set, task.spawn());
53
54    // Передача сообщений на выход компонента
55    let task = shared_tasks::mpsc_to_msgbus::MpscToMsgBus {
56        input: ch_rx_to_msgbus,
57        msg_bus: msg_bus.clone(),
58    };
59    join_set_spawn(
60        &mut task_set,
61        task.spawn().map_err(super::Error::TaskMpscToMsgBus),
62    );
63
64    // Генерирование собственного счетчика
65    let task = tasks::GenerateSelfCounter {
66        output: ch_tx_to_msgbus,
67        fn_generate_self_counter: config.fn_generate_self_counter,
68        generate_self_period: config.generate_self_period,
69    };
70    join_set_spawn(&mut task_set, task.spawn());
71
72    while let Some(result) = task_set.join_next().await {
73        result??;
74    }
75
76    Ok(())
77}
