1use tokio::task::JoinSet;
2use tracing::info;
3
4use crate::{
5    components::shared_tasks::cmp_http_client::HttpClientGeneral,
6    executor::{MsgBusLinker, join_set_spawn},
7    message::MsgDataBound,
8};
9
10use super::{Result, config, tasks};
11
12pub async fn fn_process<TMsg>(
13    msgbus_linker: MsgBusLinker<TMsg>,
14    config: config::Config<TMsg>,
15) -> Result<()>
16where
17    TMsg: MsgDataBound + 'static,
18{
19    info!("Starting http-client, configuration: {:?}", config);
20
21    let mut task_set = JoinSet::new();
22
23    let http_client_general = HttpClientGeneral {
24        msgbus_linker,
25        task_set: &mut task_set,
26        requests_input: config.requests_input,
27        requests_periodic: config.requests_periodic,
28    };
29
30    let (ch_rx_requests, ch_tx_reponse) = http_client_general.spawn();
31
32    let task = tasks::HttpClient {
33        input: ch_rx_requests,
34        output: ch_tx_reponse,
35        base_url: config.base_url,
36        timeout: config.timeout,
37    };
38    join_set_spawn(&mut task_set, "cmp_http_client", task.spawn());
39
40    while let Some(res) = task_set.join_next().await {
41        res??;
42    }
43    Ok(())
44}
