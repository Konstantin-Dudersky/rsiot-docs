34pub async fn fn_process<TMsg, TServerToClient, TClientToServer>(
35    msgbus_linker: MsgBusLinker<TMsg>,
36    config: Config<TMsg, TServerToClient, TClientToServer>,
37) -> Result<(), Error>
38where
39    TMsg: MsgDataBound + 'static,
40    TServerToClient: 'static + WebsocketMessage,
41    TClientToServer: 'static + WebsocketMessage,
42{
43    info!(
44        "Component cmp_websocket_server started. Config: {:?}",
45        config
46    );
47
48    let result = task_main(msgbus_linker, config.clone()).await;
49    match result {
50        Ok(_) => (),
51        Err(err) => error!("{:?}", err),
52    }
53    sleep(Duration::from_secs(2)).await;
54
55    Err(Error::FnProcessEnd)
56}

58async fn task_main<TMsg, TServerToClient, TClientToServer>(
59    msgbus_linker: MsgBusLinker<TMsg>,
60    config: Config<TMsg, TServerToClient, TClientToServer>,
61) -> super::Result<()>
62where
63    TMsg: MsgDataBound + 'static,
64    TServerToClient: 'static + WebsocketMessage,
65    TClientToServer: 'static + WebsocketMessage,
66{
67    let addr = format!("0.0.0.0:{}", config.port);
68
69    let listener = create_tcp_listener(addr).await?;
70
71    let cache = Arc::new(Mutex::new(HashMap::new()));
72
73    let serde_alg = SerdeAlg::new(config.serde_alg);
74
75    let (ch_tx_msgbus_to_mpsc, ch_rx_msgbus_to_mpsc) = mpsc::channel(1000);
76    let (ch_tx_input_to_clients, ch_rx_input_to_clients) = broadcast::channel(1000);
77    let (ch_tx_clients_to_output, ch_rx_clients_to_output) = mpsc::channel(1000);
78    let (ch_tx_mpsc_to_msgbus, ch_rx_mpsc_to_msgbus) = mpsc::channel(1000);
79
80    let mut task_set = JoinSet::new();
81
82    // Пересылка входящих сообщений ----------------------------------------------------------------
83    let task = shared_tasks::msgbus_to_mpsc::MsgBusToMpsc {
84        input: msgbus_linker.input(),
85        output: ch_tx_msgbus_to_mpsc,
86    };
87    join_set_spawn(
88        &mut task_set,
89        "cmp_websocket_server | msgbus_to_mpsc",
90        task.spawn().map_err(super::Error::SharedTaskMsgBusToMpsc),
91    );
92
93    // Преобразование входящих сообщений -----------------------------------------------------------
94    let task = tasks::Input {
95        input: ch_rx_msgbus_to_mpsc,
96        output: ch_tx_input_to_clients,
97        fn_input: config.fn_server_to_client,
98        cache: cache.clone(),
99    };
100    join_set_spawn(&mut task_set, "cmp_websocket_server | input", task.spawn());
101
102    // Создание исходящих сообщений ----------------------------------------------------------------
103    let task = tasks::Output {
104        input: ch_rx_clients_to_output,
105        output: ch_tx_mpsc_to_msgbus,
106        fn_output: config.fn_client_to_server,
107    };
108    join_set_spawn(&mut task_set, "cmp_websocket_server | output", task.spawn());
109
110    // Исходящие сообщения в шину сообщений --------------------------------------------------------
111    let task = shared_tasks::mpsc_to_msgbus::MpscToMsgBus {
112        input: ch_rx_mpsc_to_msgbus,
113        output: msgbus_linker.output(),
114    };
115    join_set_spawn(
116        &mut task_set,
117        "cmp_websocket_server | mpsc_to_msgbus",
118        task.spawn().map_err(super::Error::SharedTaskMpscToMsgBus),
119    );
120
121    msgbus_linker.close();
122
123    // Слушаем порт, при получении запроса создаем новое подключение WS
124    while let Ok(stream_and_addr) = listener.accept().await {
125        let ch_rx_input_to_clients = ch_rx_input_to_clients.resubscribe();
126        let ch_tx_clients_to_output = ch_tx_clients_to_output.clone();
127        let task = handle_ws_connection(
128            ch_rx_input_to_clients,
129            ch_tx_clients_to_output,
130            cache.clone(),
131            stream_and_addr,
132            serde_alg,
133        );
134        join_set_spawn(&mut task_set, "cmp_websocket_server | connection", task);
135    }
136
137    Ok(())
138}

140pub async fn handle_ws_connection<TServerToClient, TClientToServer>(
141    input: broadcast::Receiver<WsData<TServerToClient>>,
142    output: mpsc::Sender<WsData<TClientToServer>>,
143    cache: ServerToClientCache<TServerToClient>,
144    stream_and_addr: (TcpStream, SocketAddr),
145    serde_alg: SerdeAlg,
146) -> super::Result<()>
147where
148    TServerToClient: 'static + WebsocketMessage,
149    TClientToServer: 'static + WebsocketMessage,
150{
151    let addr = stream_and_addr.1;
152    let result =
153        _handle_ws_connection(input, output, cache.clone(), stream_and_addr, serde_alg).await;
154    match result {
155        Ok(_) => (),
156        Err(err) => {
157            warn!("Websocket client from address: {}, error: {}", addr, err)
158        }
159    }
160    info!("Connection closed");
161
162    Ok(())
163}

165async fn _handle_ws_connection<TServerToClient, TClientToServer>(
166    input: broadcast::Receiver<WsData<TServerToClient>>,
167    output: mpsc::Sender<WsData<TClientToServer>>,
168    cache: ServerToClientCache<TServerToClient>,
169    stream_and_addr: (TcpStream, SocketAddr),
170    serde_alg: SerdeAlg,
171) -> super::Result<()>
172where
173    TServerToClient: 'static + WebsocketMessage,
174    TClientToServer: 'static + WebsocketMessage,
175{
176    info!("Incoming TCP connection from: {}", stream_and_addr.1);
177    let ws_stream = accept_async(stream_and_addr.0).await?;
178    let (websocket_write, websocket_read) = ws_stream.split();
179    info!("WebSocket connection established: {:?}", stream_and_addr.1);
180
181    let client_id = stream_and_addr.1.to_string();
182
183    let mut task_set: JoinSet<super::Result<()>> = JoinSet::new();
184
185    // Отправление данных клиенту
186    let task = tasks::SendToClient {
187        input: input.resubscribe(),
188        websocket_write,
189        cache: cache.clone(),
190        serde_alg,
191        client_id: client_id.clone(),
192    };
193    join_set_spawn(
194        &mut task_set,
195        "cmp_websocket_server | send_to_client",
196        task.spawn(),
197    );
198
199    // Получение данных от клиента
200    let task = tasks::RcvFromClient {
201        output: output.clone(),
202        websocket_read,
203        serde_alg,
204        client_id: client_id.clone(),
205    };
206    join_set_spawn(
207        &mut task_set,
208        "cmp_websocket_server | recv_from_client",
209        task.spawn(),
210    );
211
212    while let Some(res) = task_set.join_next().await {
213        let err = match res {
214            Ok(val) => match val {
215                Ok(_) => continue,
216                Err(err) => format!("{err}"),
217            },
218            Err(err) => format!("{err}"),
219        };
220        warn!("Connection error: {}", err);
221        task_set.shutdown().await;
222    }
223    Ok(())
224}

226async fn create_tcp_listener(addr: String) -> super::Result<TcpListener> {
227    let listener = TcpListener::bind(&addr).await;
228    let listener = listener.map_err(Error::BindToPort)?;
229    info!("Listening on: {}", addr);
230    Ok(listener)
