20pub async fn fn_process<TMsg>(
21    config: Config<TMsg>,
22    msgbus_linker: MsgBusLinker<TMsg>,
23) -> super::Result<()>
24where
25    TMsg: 'static + MsgDataBound,
26{
27    let mut task_set = JoinSet::new();
28
29    let config_fn_process_master = FieldbusExecution {
30        msgbus_linker,
31        task_set: &mut task_set,
32        error_filter: Error::TaskFilter,
33        error_mpsc_to_msgbus: Error::TaskMpscToMsgBus,
34        error_master_device: Error::DeviceError,
35        error_tokiompscsend: || Error::TokioSyncMpsc,
36        devices: config.devices,
37    };
38    let (ch_rx_devices_to_fieldbus, ch_tx_fieldbus_to_devices) = config_fn_process_master.spawn();
39
40    let task = I2cComm {
41        input: ch_rx_devices_to_fieldbus,
42        output: ch_tx_fieldbus_to_devices,
43        dev_i2c: config.dev_i2c,
44    };
45    join_set_spawn(
46        &mut task_set,
47        "cmp_linux_i2c_master | i2c_comm",
48        task.spawn(),
49    );
50
51    while let Some(res) = task_set.join_next().await {
52        res??
53    }
54
55    Ok(())
56}

58pub struct I2cComm {
59    pub input: mpsc::Receiver<FieldbusRequestWithIndex<i2c_master::FieldbusRequest>>,
60    pub output: mpsc::Sender<FieldbusResponseWithIndex<i2c_master::FieldbusResponse>>,
61    pub dev_i2c: String,
62}
63impl I2cComm {
64    pub async fn spawn(mut self) -> super::Result<()> {
65        let mut bus = LinuxI2CBus::new(self.dev_i2c.clone())?;
66
67        while let Some(fieldbus_request) = self.input.recv().await {
68            trace!("New i2c request: {:?}", fieldbus_request);
69
70            let device_index = fieldbus_request.device_index;
71            let request = fieldbus_request.request;
72
73            // Выполняем все операции в цикле
74            let response_payload = {
75                // Ответы от слейва
76                let mut responses = vec![];
77                let mut error = "".to_string();
78
79                for operation in request.operations {
80                    let response = make_i2c_operation(&mut bus, request.address, &operation).await;
81                    let response = match response {
82                        Ok(response) => response,
83                        Err(err) => {
84                            error = err.to_string();
85                            break;
86                        }
87                    };
88                    responses.push(response);
89                }
90
91                if error.is_empty() {
92                    Ok(responses)
93                } else {
94                    Err(error)
95                }
96            };
97
98            let response = i2c_master::FieldbusResponse {
99                request_creation_time: request.request_creation_time,
100                request_duration: request.request_creation_time.elapsed(),
101                request_kind: request.request_kind,
102                payload: response_payload,
103            };
104            let response_with_index = FieldbusResponseWithIndex {
105                device_index,
106                response,
107            };
108
109            trace!("I2C response: {:?}", response_with_index);
110            self.output
111                .send(response_with_index)
112                .await
113                .map_err(|_| Error::TokioSyncMpsc)?;
114        }
115
116        Ok(())
117    }
118}

120/// Выполняем обмен данными
121///
122/// Возвращаем ответ в виде вектора байтов
123async fn make_i2c_operation(
124    bus: &mut LinuxI2CBus,
125    i2c_address: I2cAddress,
126    operation: &i2c_master::Operation,
127) -> Result<Vec<u8>, LinuxI2CError> {
128    // Определяем адрес
129    let address = match i2c_address {
130        I2cAddress::Direct { address } => address as u16,
131        I2cAddress::Mux {
132            mux_address,
133            channel,
134            address,
135        } => {
136            // Открываем канал на мультиплексоре
137            let mux_data = [channel];
138            let mut transaction =
139                [LinuxI2CMessage::write(&mux_data).with_address(mux_address as u16)];
140            bus.transfer(&mut transaction)?;
141
142            address as u16
143        }
144    };
145
146    let read_result = match operation {
147        i2c_master::Operation::Delay { delay } => {
148            sleep(*delay).await;
149            Ok(vec![])
150        }
151
152        i2c_master::Operation::WriteRead {
153            write_data,
154            read_size,
155        } => {
156            let mut read_data = vec![0; *read_size as usize];
157
158            let mut transaction = [
159                LinuxI2CMessage::write(write_data).with_address(address),
160                LinuxI2CMessage::read(&mut read_data).with_address(address),
161            ];
162
163            bus.transfer(&mut transaction)?;
164            trace!("Read data: {:x?}", read_data);
165            Ok(read_data)
166        }
167
168        i2c_master::Operation::Write { write_data } => {
169            let mut transaction = [LinuxI2CMessage::write(write_data).with_address(address)];
170            bus.transfer(&mut transaction)?;
171            Ok(vec![])
172        }
173
174        i2c_master::Operation::Read { read_size } => {
175            let mut read_data = vec![0; *read_size as usize];
176
177            let mut transaction = [LinuxI2CMessage::read(&mut read_data).with_address(address)];
178            bus.transfer(&mut transaction)?;
179            trace!("Read data: {:x?}", read_data);
180            Ok(read_data)
181        }
182    };
183
184    // Закрываем все каналы на мультиплексоре
185    match i2c_address {
186        I2cAddress::Direct { .. } => (),
187        I2cAddress::Mux { mux_address, .. } => {
188            let mut transaction =
189                [LinuxI2CMessage::write(&[0x00]).with_address(mux_address as u16)];
190            bus.transfer(&mut transaction)?;
191        }
192    };
193
194    read_result
195}
