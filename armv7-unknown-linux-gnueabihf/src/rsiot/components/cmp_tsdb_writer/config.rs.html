42/// Строка подключения к БД
43#[derive(Clone, Debug)]
44pub struct ConfigConnectionString {
45    /// Пользователь
46    ///
47    /// `postgres`
48    pub user: String,
49
50    /// Пароль
51    ///
52    /// `postgres`
53    pub password: String,
54
55    /// Хост
56    ///
57    /// `localhost`
58    pub database_host: String,
59
60    /// Порт
61    ///
62    /// `5432`
63    pub port: u16,
64
65    /// Проект
66    pub prj: String,
67
68    /// Хост
69    pub hst: String,
70
71    /// Сервис
72    pub svc: String,
73}
74impl ConfigConnectionString {
75    /// Возвращает строку подключения к БД
76    pub fn connection_string(&self) -> String {
77        format!(
78            "postgres://{}:{}@{}:{}/{}__{}__{}",
79            self.user, self.password, self.database_host, self.port, self.prj, self.hst, self.svc
80        )
81    }
82}

84/// Конфигурация таблицы для сохранения данных в БД
85#[derive(Clone, Debug)]
86pub struct ConfigTable<TMsg>
87where
88    TMsg: MsgDataBound,
89{
90    /// Компонент
91    pub cmp: String,
92
93    /// Тег
94    pub tag: String,
95
96    /// Разделение на части по времени
97    pub chunk_interval: Duration,
98
99    /// Сжимать данные через указанный интервал после записи
100    pub compress_interval: Duration,
101
102    /// Удалять данные старше указанного интервала. Если не указано, то данные не удаляются
103    pub retention_interval: Option<Duration>,
104
105    /// Поля таблицы
106    pub fields: Vec<ConfigTableField>,
107
108    /// Функция преобразования сообщений в строки для Timescaledb
109    pub fn_input: fn(&TMsg) -> Result<Option<String>, Error>,
110
111    /// Удалить таблицу перед записью
112    pub delete_before_write: bool,
113}
114impl<TMsg> ConfigTable<TMsg>
115where
116    TMsg: MsgDataBound,
117{
118    /// Название таблицы в БД
119    pub fn table_name(&self) -> String {
120        format!("{}__{}", self.cmp, self.tag)
121    }
122}

124/// Конфигурация поля таблицы
125#[derive(Clone, Debug)]
126pub struct ConfigTableField {
127    /// Название поля таблицы
128    pub field_name: String,
129
130    /// Тип поля таблицы
131    pub data_type: ConfigTableFieldType,
132}
133
134/// Тип поля таблицы
135#[derive(Clone, Debug)]
136pub enum ConfigTableFieldType {
137    /// Произвольный текст
138    CharacterText,
139
140    /// 4 байта
141    NumericInteger,
142
143    /// 8 байт
144    NumericDoublePrecision,
145
146    /// 1 байт
147    Boolean,
148}
149impl ConfigTableFieldType {
150    /// Преобразование типа поля в тип PostgreSQL
151    pub fn into_pg_type(&self) -> &'static str {
152        match self {
153            Self::CharacterText => "TEXT",
154            Self::NumericInteger => "INTEGER",
155            Self::NumericDoublePrecision => "DOUBLE PRECISION",
156            Self::Boolean => "BOOLEAN",
157        }
158    }
159}

161pub(crate) struct ConfigTableForSetup {
162    pub table_name: String,
163    pub delete_before_write: bool,
164    pub chunk_interval: Duration,
165    pub compress_interval: Duration,
166    pub retention_interval: Option<Duration>,
167    pub values: Vec<ConfigTableField>,
168}
169
170impl<TMsg> From<&ConfigTable<TMsg>> for ConfigTableForSetup
171where
172    TMsg: MsgDataBound,
173{
174    fn from(config_table: &ConfigTable<TMsg>) -> Self {
175        ConfigTableForSetup {
176            table_name: config_table.table_name(),
177            delete_before_write: config_table.delete_before_write,
178            chunk_interval: config_table.chunk_interval,
179            compress_interval: config_table.compress_interval,
180            retention_interval: config_table.retention_interval,
181            values: config_table.fields.clone(),
182        }
183    }
184}
