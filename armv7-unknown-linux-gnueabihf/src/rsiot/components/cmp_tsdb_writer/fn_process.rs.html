1use std::{sync::Arc, time::Duration};
2
3use tokio::{
4    sync::{Mutex, mpsc},
5    task::{JoinHandle, JoinSet},
6};
7use tracing::info;
8
9use crate::{
10    executor::{MsgBusLinker, join_set_spawn},
11    message::MsgDataBound,
12};
13
14use super::{COMPONENT_NAME, Error, QueryStat, config::Config, config::ConfigTableForSetup, tasks};
15
16pub async fn fn_process<TMsg>(
17    msgbus_linker: MsgBusLinker<TMsg>,
18    config: Config<TMsg>,
19) -> Result<(), Error>
20where
21    TMsg: 'static + MsgDataBound,
22{
23    info!("Start {COMPONENT_NAME}");
24
25    let (ch_tx_database_to_results, ch_rx_database_to_results) =
26        mpsc::channel::<JoinHandle<Result<QueryStat, Error>>>(10);
27
28    let mut task_set = JoinSet::new();
29
30    let db_pool_mutex = Arc::new(Mutex::new(None));
31
32    let tables_setup: Vec<ConfigTableForSetup> = config.tables.iter().map(|t| t.into()).collect();
33    let task = tasks::SetupDatabase {
34        connection_string: config.connection_string.clone(),
35        reconnect_interval: Duration::from_millis(1_000),
36        tables: tables_setup,
37        db_pool_mutex: db_pool_mutex.clone(),
38        max_connections: config.max_connections,
39    };
40    join_set_spawn(
41        &mut task_set,
42        format!("{COMPONENT_NAME} | setup_database"),
43        task.spawn(),
44    );
45
46    for table in config.tables {
47        let (tx_input_to_sql, rx_input_to_sql) = mpsc::channel::<tasks::InnerMessage>(1000);
48
49        let table_name = table.table_name();
50
51        let task = tasks::Input {
52            msgbus_input: msgbus_linker.input(),
53            output: tx_input_to_sql.clone(),
54            table,
55        };
56        join_set_spawn(
57            &mut task_set,
58            format!("{COMPONENT_NAME} | input | {}", table_name),
59            task.spawn(),
60        );
61
62        let task = tasks::Periodic {
63            output: tx_input_to_sql,
64            save_by_period: config.save_by_period,
65        };
66        join_set_spawn(
67            &mut task_set,
68            format!("{COMPONENT_NAME} | periodic | {}", table_name),
69            task.spawn(),
70        );
71
72        let task = tasks::PrepareSQL {
73            input: rx_input_to_sql,
74            output: ch_tx_database_to_results.clone(),
75            table_name: table_name.clone(),
76            db_pool_mutex: db_pool_mutex.clone(),
77            save_by_period: config.save_by_period,
78            save_by_row_count: config.save_by_row_count,
79        };
80        join_set_spawn(
81            &mut task_set,
82            format!("{COMPONENT_NAME} | prepare_sql | {}", table_name),
83            task.spawn(),
84        );
85    }
86
87    msgbus_linker.close();
88
89    let task = tasks::WaitResult {
90        input: ch_rx_database_to_results,
91        fn_query_stat: config.fn_query_stat,
92    };
93    join_set_spawn(
94        &mut task_set,
95        format!("{COMPONENT_NAME} | wait_result"),
96        task.spawn(),
97    );
98
99    while let Some(res) = task_set.join_next().await {
100        res??;
101    }
102
103    Ok(())
104}
