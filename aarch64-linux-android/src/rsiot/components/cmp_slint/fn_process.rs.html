1use std::time::Duration;
2
3use futures::TryFutureExt;
4use slint::ComponentHandle;
5use tokio::{sync::mpsc, task::JoinSet};
6
7use crate::{
8    components::shared_tasks,
9    executor::{MsgBusLinker, join_set_spawn},
10    message::MsgDataBound,
11};
12
13use super::{Config, Error, Result, tasks};
14
15pub async fn fn_process<TMainWindow, TMsg>(
16    config: Config<TMsg, TMainWindow>,
17    msgbus_linker: MsgBusLinker<TMsg>,
18) -> Result<()>
19where
20    TMsg: MsgDataBound + 'static,
21    TMainWindow: ComponentHandle + 'static,
22{
23    let buffer_size = msgbus_linker.max_capacity();
24
25    let (ch_tx_output_to_filter, ch_rx_output_to_filter) = mpsc::channel(buffer_size);
26    let (ch_tx_filter_to_msgbus, ch_rx_filter_to_msgbus) = mpsc::channel(buffer_size);
27
28    let mut task_set = JoinSet::new();
29
30    // Обработка входящих сообщений и изменение данных в приложении Slint
31    let task = tasks::Input {
32        input: msgbus_linker.input(),
33        slint_window: config.slint_window.clone(),
34        fn_input: config.fn_input,
35    };
36    join_set_spawn(&mut task_set, "cmp_slint | input", task.spawn());
37
38    // Создание сообщений на основе взаимодествия с приложением Slint
39    let task = tasks::Output {
40        output: ch_tx_output_to_filter,
41        slint_window: config.slint_window.clone(),
42        fn_output: config.fn_output,
43    };
44    join_set_spawn(&mut task_set, "cmp_slint | output", task.spawn());
45
46    // Фильтрация сообещений
47    let task = shared_tasks::filter_send_periodically::FilterSendPeriodically {
48        input: ch_rx_output_to_filter,
49        output: ch_tx_filter_to_msgbus,
50        period: Duration::from_millis(100),
51    };
52    join_set_spawn(
53        &mut task_set,
54        "cmp_slint | filter",
55        task.spawn().map_err(Error::TaskFilterSendPeriodically),
56    );
57
58    // Передача сообщений в шину сообщений
59    let task = shared_tasks::mpsc_to_msgbus::MpscToMsgBus {
60        input: ch_rx_filter_to_msgbus,
61        output: msgbus_linker.output(),
62    };
63    join_set_spawn(
64        &mut task_set,
65        "cmp_slint | mpsc_to_msgbus",
66        task.spawn().map_err(Error::TaskMpscToMsgBus),
67    );
68
69    msgbus_linker.close();
70
71    while let Some(res) = task_set.join_next().await {
72        res??;
73    }
74    Ok(())
75}
