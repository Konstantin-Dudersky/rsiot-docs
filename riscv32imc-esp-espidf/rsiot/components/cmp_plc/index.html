use rsiot::components::cmp_plc::plc::types::Resettable;

use crate::msg::*;

use super::logic::fb_main;

pub fn fn_input(input: &mut fb_main::I, msg: &Msg) {
    match msg {
        Msg::Filesystem(_) => (),
        Msg::Plc(_) => (),
        Msg::MsgInjectPeriodic(msg) => match msg {
            MsgInjectPeriodic::Counter(v) => {
                input.counter = *v;
                input.resettabe_bit = Resettable::new(true);
            }
        },
    }
}

§cfg_plc/retention.rs

use std::time::Duration;

use rsiot::{components::cmp_plc::*, message::Message};

use crate::msg::*;

use super::logic::fb_main;

pub fn retention() -> ConfigRetention<Msg, fb_main::I, fb_main::Q, fb_main::S> {
    ConfigRetention {
        save_period: Duration::from_secs(5),
        fn_export: |input, output, stat| {
            let msg_input = Message::new_custom(Msg::Plc(Plc::Input(input.clone())));
            let msg_output = Message::new_custom(Msg::Plc(Plc::Output(output.clone())));
            let msg_static = Message::new_custom(Msg::Plc(Plc::Static(stat.clone())));

            Some(vec![msg_input, msg_output, msg_static])
        },
        fn_import_static: |msg| {
            let msg = msg.get_custom_data();
            let Some(msg) = msg else { return Ok(None) };
            match msg {
                Msg::Filesystem(Filesystem::MemoryStatic(stat)) => Ok(Some(stat)),
                _ => Ok(None),
            }
        },
        restore_timeout: Duration::from_secs(2),
    }
}

use rsiot::components::cmp_plc::plc::{
    FbSystemData, FunctionBlockBase, IFunctionBlock, types::Resettable,
};
use serde::{Deserialize, Serialize};

pub type FB = FunctionBlockBase<I, Q, S>;

impl IFunctionBlock<I, Q, S> for FunctionBlockBase<I, Q, S> {
    fn logic(input: &mut I, stat: &mut S, system_data: &FbSystemData) -> Q {
        logic(input, stat, system_data)
    }
}

/// Input ------------------------------------------------------------------------------------------
#[derive(Clone, Debug, Default, Deserialize, PartialEq, Serialize)]
pub struct I {
    pub counter: u32,
    pub resettabe_bit: Resettable<bool>,
}

/// Output -----------------------------------------------------------------------------------------
#[derive(Clone, Debug, Default, Deserialize, PartialEq, Serialize)]
pub struct Q {}

/// Static -----------------------------------------------------------------------------------------
#[derive(Clone, Debug, Default, Deserialize, PartialEq, Serialize)]
pub struct S {
    pub fb_simple: super::fb_simple::FB,
}

/// Логика -----------------------------------------------------------------------------------------
pub fn logic(_input: &mut I, _stat: &mut S, _system_data: &FbSystemData) -> Q {
    Q {}
}
