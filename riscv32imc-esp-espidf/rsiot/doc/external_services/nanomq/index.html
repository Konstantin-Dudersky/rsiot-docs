Настройка NanoMQ

§docker

services:
  service_nanomq:
    container_name: service_nanomq
    hostname: service_nanomq
    image: emqx/nanomq:0.24 # https://hub.docker.com/r/emqx/nanomq/tags
    networks:
      - network_internal
    ports:
      - "1883:1883" # mqtt
      - "8083:8083" # mqtt over websocket
      - "8883:8883" # mqtt over tls
    profiles:
      - dev
      - target
    restart: always
