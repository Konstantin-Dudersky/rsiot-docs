17pub(crate) struct MqttGeneralTasks<'a, TMsg, TError>
18where
19    TMsg: MsgDataBound,
20{
21    /// Шина сообщений
22    pub msg_bus: MsgBusLinker<TMsg>,
23
24    /// Ёмкость очередей сообщений между задачами
25    pub buffer_size: usize,
26
27    /// Ссылка на коллекцию задач tokio
28    pub task_set: &'a mut JoinSet<Result<(), TError>>,
29
30    /// Настройка публикации данных в брокере
31    pub publish: ConfigPublish<TMsg>,
32
33    /// Настройка подписки на данные из брокера
34    pub subscribe: ConfigSubscribe<TMsg>,
35
36    /// Генератор сообщений MQTT
37    pub mqtt_msg_gen: MqttMsgGen,
38
39    pub error_fn_publish: fn(anyhow::Error) -> TError,
40    pub error_fn_subscribe: fn(anyhow::Error) -> TError,
41
42    pub error_task_end_input: fn() -> TError,
43
44    pub error_task_end_output: fn() -> TError,
45
46    pub error_tokio_mpsc_send: fn() -> TError,
47}
48
49impl<TMsg, TError> MqttGeneralTasks<'_, TMsg, TError>
50where
51    TMsg: 'static + MsgDataBound,
52    TError: 'static + Send,
53{
54    pub fn spawn(self) -> (mpsc::Receiver<MqttMsgSend>, mpsc::Sender<MqttMsgRecv>) {
55        let (ch_tx_send, ch_rx_send) = mpsc::channel::<MqttMsgSend>(self.buffer_size);
56        let (ch_tx_recv, ch_rx_recv) = mpsc::channel::<MqttMsgRecv>(self.buffer_size);
57
58        // Получение сообщений из шины
59        let task = Input {
60            input: self.msg_bus.input(),
61            output: ch_tx_send.clone(),
62            config_publish: self.publish,
63            mqtt_msg_gen: self.mqtt_msg_gen.clone(),
64            error_fn_publish: self.error_fn_publish,
65            error_task_end: self.error_task_end_input,
66            error_tokio_mpsc_send: self.error_tokio_mpsc_send,
67        };
68        join_set_spawn(self.task_set, "mqtt_general_tasks | input", task.spawn());
69
70        let task = Output {
71            input: ch_rx_recv,
72            output_send: ch_tx_send,
73            output_msg_bus: self.msg_bus.output(),
74            config_subscribe: self.subscribe,
75            mqtt_msg_gen: self.mqtt_msg_gen.clone(),
76            error_fn_subscribe: self.error_fn_subscribe,
77            error_tokio_mpsc_send: self.error_tokio_mpsc_send,
78            error_task_end: self.error_task_end_output,
79        };
80        join_set_spawn(self.task_set, "mqtt_general_tasks | output", task.spawn());
81
82        (ch_rx_send, ch_tx_recv)
83    }
