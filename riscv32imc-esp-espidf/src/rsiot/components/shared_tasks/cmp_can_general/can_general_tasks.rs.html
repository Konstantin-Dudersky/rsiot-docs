17pub(crate) struct CanGeneralTasks<'a, TMsg, TBuffer, TError>
18where
19    TMsg: MsgDataBound,
20    TBuffer: BufferBound,
21{
22    /// Подключение к шине сообщений
23    pub msgbus_linker: MsgBusLinker<TMsg>,
24
25    /// Значение в буфере по умолчанию.
26    ///
27    /// Буфер используется для отправки периодических сообщений.
28    ///
29    /// Если буфер не используется, можно задать значение `()`.
30    pub buffer_default: TBuffer,
31
32    /// Ссылка на коллекцию задач tokio
33    pub task_set: &'a mut JoinSet<Result<(), TError>>,
34
35    pub fn_input: fn(&TMsg, &mut TBuffer) -> anyhow::Result<Option<Vec<CanFrame>>>,
36
37    pub period: Duration,
38
39    pub fn_periodic: fn(&TBuffer) -> anyhow::Result<Option<Vec<CanFrame>>>,
40
41    pub fn_output: fn(CanFrame) -> Option<Vec<TMsg>>,
42
43    pub error_task_end_input: fn() -> TError,
44
45    pub error_task_end_output: fn() -> TError,
46
47    pub error_tokio_mpsc_send: fn() -> TError,
48}

50impl<TMsg, TBuffer, TError> CanGeneralTasks<'_, TMsg, TBuffer, TError>
51where
52    TMsg: 'static + MsgDataBound,
53    TBuffer: 'static + BufferBound,
54    TError: 'static + Send,
55{
56    pub fn spawn(self) -> (mpsc::Receiver<CanFrame>, mpsc::Sender<CanFrame>) {
57        let buffer = Arc::new(Mutex::new(self.buffer_default));
58
59        let buffer_size = self.msgbus_linker.max_capacity();
60
61        let (ch_tx_send_to_can, ch_rx_send_to_can) = mpsc::channel::<CanFrame>(buffer_size);
62        let (ch_tx_recv_from_can, ch_rx_recv_from_can) = mpsc::channel::<CanFrame>(buffer_size);
63
64        // Получение сообщений из шины
65        let task = Input {
66            input: self.msgbus_linker.input(),
67            output: ch_tx_send_to_can.clone(),
68            buffer: buffer.clone(),
69            fn_input: self.fn_input,
70            error_task_end: self.error_task_end_input,
71            error_tokio_mpsc_send: self.error_tokio_mpsc_send,
72        };
73        join_set_spawn(self.task_set, "can_general_tasks | input", task.spawn());
74
75        let task = Periodic {
76            output: ch_tx_send_to_can,
77            buffer,
78            period: self.period,
79            fn_periodic: self.fn_periodic,
80            error_tokio_mpsc_send: self.error_tokio_mpsc_send,
81        };
82        join_set_spawn(self.task_set, "can_general_tasks | periodic", task.spawn());
83
84        let task = Output {
85            input: ch_rx_recv_from_can,
86            output: self.msgbus_linker.output(),
87            fn_output: self.fn_output,
88            error_task_end: self.error_task_end_output,
89            error_tokio_mpsc_send: self.error_tokio_mpsc_send,
90        };
91        join_set_spawn(self.task_set, "can_general_tasks | output", task.spawn());
92
93        self.msgbus_linker.close();
94
95        (ch_rx_send_to_can, ch_tx_recv_from_can)
96    }
