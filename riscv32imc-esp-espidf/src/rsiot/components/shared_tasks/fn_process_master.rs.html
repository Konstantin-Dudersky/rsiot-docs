23/// Запуск задач, общих для всех компонентов, выполняющих опрос устройств по шине
24pub struct FnProcessMaster<'a, TMsg, TError, TFieldbusRequest, TFieldbusResponse>
25where
26    TMsg: MsgDataBound + 'static,
27    TError: Send + Sync + 'static,
28{
29    /// Подключение к шине MsgBus
30    pub msgbus_linker: MsgBusLinker<TMsg>,
31
32    /// Ссылка на коллекцию задач tokio
33    pub task_set: &'a mut JoinSet<Result<(), TError>>,
34
35    /// Ошибка filter_identical_data
36    pub error_filter: fn(filter_identical_data::Error) -> TError,
37
38    /// Ошибка mpsc_to_msgbus
39    pub error_mpsc_to_msgbus: fn(mpsc_to_msgbus::Error) -> TError,
40
41    /// Ошибка master_device
42    pub error_master_device: fn(master_device::Error) -> TError,
43
44    /// Ошибка tokio::mpsc::send
45    pub error_tokiompscsend: fn() -> TError,
46
47    /// Массив устройств
48    pub devices: Vec<Box<dyn DeviceTrait<TMsg, TFieldbusRequest, TFieldbusResponse>>>,
49}
50
51impl<TMsg, TError, TFieldbusRequest, TFieldbusResponse>
52    FnProcessMaster<'_, TMsg, TError, TFieldbusRequest, TFieldbusResponse>
53where
54    TMsg: MsgDataBound + 'static,
55    TError: Send + Sync + 'static,
56    TFieldbusRequest: RequestResponseBound + 'static,
57    TFieldbusResponse: RequestResponseBound + 'static,
58{
59    /// Запуск задач.
60    ///
61    /// Возвращает кортеж с каналами передачи запросов / ответов
62    pub fn spawn(
63        self,
64    ) -> (
65        mpsc::Receiver<FieldbusRequestWithIndex<TFieldbusRequest>>,
66        mpsc::Sender<FieldbusResponseWithIndex<TFieldbusResponse>>,
67    ) {
68        let devices_count = self.devices.len();
69        let buffer_size = self.msgbus_linker.max_capacity();
70
71        // Создание каналов передачи данных --------------------------------------------------------
72
73        // Каналы передачи запросов из устройств в задачи добавления индекса
74        let mut ch_tx_device_to_addindex = vec![];
75        let mut ch_rx_device_to_addindex = vec![];
76
77        for _ in 0..devices_count {
78            let (ch_tx, ch_rx) = mpsc::channel::<TFieldbusRequest>(buffer_size);
79            ch_tx_device_to_addindex.push(ch_tx);
80            ch_rx_device_to_addindex.push(ch_rx);
81        }
82
83        // Канал передачи запросов из задач добавления индекса в шину
84        let (ch_tx_addindex_to_fieldbus, ch_rx_addindex_to_fieldbus) =
85            mpsc::channel::<FieldbusRequestWithIndex<TFieldbusRequest>>(buffer_size);
86
87        // Канал передачи ответов из шины в задачу разделения ответов для устройств
88        let (ch_tx_fieldbus_to_split, ch_rx_fieldbus_to_split) =
89            mpsc::channel::<FieldbusResponseWithIndex<TFieldbusResponse>>(buffer_size);
90
91        // Каналы передачи ответов из задачи разделения на устройства
92        let mut ch_tx_split_to_devices = vec![];
93        let mut ch_rx_split_to_devices = vec![];
94        for _ in 0..devices_count {
95            let (ch_tx, ch_rx) = mpsc::channel::<TFieldbusResponse>(buffer_size);
96            ch_tx_split_to_devices.push(ch_tx);
97            ch_rx_split_to_devices.push(Some(ch_rx));
98        }
99
100        // Канал передачи сообщений из устройств на фильтр
101        let (ch_tx_devices_to_filter, ch_rx_devices_to_filter) =
102            mpsc::channel::<Message<TMsg>>(buffer_size);
103
104        // Канал передачи сообщений из фильтра на выход компонента
105        let (ch_tx_filter_to_msgbus, ch_rx_filter_to_msgbus) =
106            mpsc::channel::<Message<TMsg>>(buffer_size);
107
108        // Задачи выполнения устройств -------------------------------------------------------------
109        let mut input_vec = vec![];
110        for _ in 0..self.devices.len() {
111            input_vec.push(self.msgbus_linker.input());
112        }
113
114        for (index, device) in self.devices.into_iter().enumerate() {
115            let ch_rx_msgbus_to_devices = input_vec[index].clone();
116            let ch_tx_device_to_addindex = ch_tx_device_to_addindex[index].clone();
117            let Some(ch_rx_fieldbus_to_device) = ch_rx_split_to_devices[index].take() else {
118                panic!("Error configuration in fn_process_master");
119            };
120            let ch_tx_devices_to_filter = ch_tx_devices_to_filter.clone();
121            let task = device.spawn(
122                ch_rx_msgbus_to_devices,
123                ch_tx_device_to_addindex,
124                ch_rx_fieldbus_to_device,
125                ch_tx_devices_to_filter,
126            );
127            join_set_spawn(
128                self.task_set,
129                "fn_process_master | device",
130                task.map_err(self.error_master_device),
131            );
132        }
133
134        // Задачи добавления индекса
135        for (device_index, ch_rx) in ch_rx_device_to_addindex.into_iter().enumerate() {
136            let task = AddIndex {
137                input: ch_rx,
138                output: ch_tx_addindex_to_fieldbus.clone(),
139                device_index,
140                error_tokiompscsend: self.error_tokiompscsend,
141            };
142            join_set_spawn(self.task_set, "fn_process_master | add_index", task.spawn());
143        }
144
145        // Задача разделения ответов
146        let task = SplitResponses {
147            input: ch_rx_fieldbus_to_split,
148            output: ch_tx_split_to_devices,
149            error_tokiompscsend: self.error_tokiompscsend,
150        };
151        join_set_spawn(
152            self.task_set,
153            "fn_process_master | split_responses",
154            task.spawn(),
155        );
156
157        // Фильтрация одинаковых сообщений ---------------------------------------------------------
158        let task = filter_identical_data::FilterIdenticalData {
159            input: ch_rx_devices_to_filter,
160            output: ch_tx_filter_to_msgbus,
161        };
162        join_set_spawn(
163            self.task_set,
164            "fn_process_master | filter_identical_data",
165            task.spawn().map_err(self.error_filter),
166        );
167
168        // Создаем исходящие сообщения -------------------------------------------------------------
169        let task = mpsc_to_msgbus::MpscToMsgBus {
170            input: ch_rx_filter_to_msgbus,
171            output: self.msgbus_linker.output(),
172        };
173        join_set_spawn(
174            self.task_set,
175            "fn_process_master | mpsc_to_msgbus",
176            task.spawn().map_err(self.error_mpsc_to_msgbus),
177        );
178
179        drop(self.msgbus_linker);
180
181        (ch_rx_addindex_to_fieldbus, ch_tx_fieldbus_to_split)
182    }
183}

185struct AddIndex<TFieldbusRequest, TError>
186where
187    TFieldbusRequest: RequestResponseBound,
188{
189    pub input: mpsc::Receiver<TFieldbusRequest>,
190    pub output: mpsc::Sender<FieldbusRequestWithIndex<TFieldbusRequest>>,
191    pub device_index: usize,
192    pub error_tokiompscsend: fn() -> TError,
193}
194impl<TFieldbusRequest, TError> AddIndex<TFieldbusRequest, TError>
195where
196    TFieldbusRequest: RequestResponseBound,
197{
198    pub async fn spawn(mut self) -> Result<(), TError> {
199        while let Some(request) = self.input.recv().await {
200            let request_with_index = FieldbusRequestWithIndex {
201                device_index: self.device_index,
202                request,
203            };
204            self.output
205                .send(request_with_index)
206                .await
207                .map_err(|_| (self.error_tokiompscsend)())?;
208        }
209        Ok(())
210    }
211}

213struct SplitResponses<TFieldbusResponse, TError>
214where
215    TFieldbusResponse: RequestResponseBound,
216{
217    pub input: mpsc::Receiver<FieldbusResponseWithIndex<TFieldbusResponse>>,
218    pub output: Vec<mpsc::Sender<TFieldbusResponse>>,
219    pub error_tokiompscsend: fn() -> TError,
220}
221impl<TFieldbusResponse, TError> SplitResponses<TFieldbusResponse, TError>
222where
223    TFieldbusResponse: RequestResponseBound,
224{
225    pub async fn spawn(mut self) -> Result<(), TError> {
226        while let Some(response_with_index) = self.input.recv().await {
227            let device_index = response_with_index.device_index;
228            let response = response_with_index.response;
229            self.output[device_index]
230                .send(response)
231                .await
232                .map_err(|_| (self.error_tokiompscsend)())?;
233        }
234        Ok(())
235    }
