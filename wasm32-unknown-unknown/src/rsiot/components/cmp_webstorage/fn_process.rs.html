1use futures::TryFutureExt;
2use tokio::{sync::mpsc::channel, task::JoinSet};
3
4use crate::{
5    components::shared_tasks,
6    executor::{join_set_spawn, CmpInOut},
7    message::*,
8};
9
10use super::{tasks, Config};
11
12pub async fn fn_process<TMsg>(config: Config<TMsg>, msg_bus: CmpInOut<TMsg>) -> super::Result<()>
13where
14    TMsg: MsgDataBound + 'static,
15{
16    let mut task_set: JoinSet<super::Result<()>> = JoinSet::new();
17
18    let channel_capacity = msg_bus.max_capacity();
19
20    let (ch_tx_msgbus_to_input, ch_rx_msgbus_to_input) = channel(channel_capacity);
21    let (ch_tx_input_to_output, ch_rx_input_to_output) = channel(channel_capacity);
22    let (ch_tx_output_to_msgbus, ch_rx_output_to_msgbus) = channel(channel_capacity);
23
24    // Со входа компонента на задачу Input
25    let task_0 = shared_tasks::msgbus_to_mpsc::MsgBusToMpsc {
26        msg_bus: msg_bus.clone(),
27        output: ch_tx_msgbus_to_input,
28    };
29    join_set_spawn(
30        &mut task_set,
31        task_0.spawn().map_err(super::Error::TaskMsgBusToMpsc),
32    );
33
34    // Обработка входящих сообщений
35    let task_1 = tasks::Input {
36        input: ch_rx_msgbus_to_input,
37        output: ch_tx_input_to_output,
38        storage_kind: config.storage_kind,
39        fn_input: config.fn_input,
40    };
41    join_set_spawn(&mut task_set, task_1.spawn());
42
43    // Загрузка значений из хранилища и отправка исходящих сообщений
44    let task_2 = tasks::Output {
45        input: ch_rx_input_to_output,
46        output: ch_tx_output_to_msgbus,
47        storage_kind: config.storage_kind,
48        default_messages: config.default_messages,
49        fn_output: config.fn_output,
50    };
51    join_set_spawn(&mut task_set, task_2.spawn());
52
53    // Отправка исходящих сообщений
54    let task_3 = shared_tasks::mpsc_to_msgbus::MpscToMsgBus {
55        input: ch_rx_output_to_msgbus,
56        msg_bus: msg_bus.clone(),
57    };
58    join_set_spawn(
59        &mut task_set,
60        task_3.spawn().map_err(super::Error::TaskMpscToMsgBus),
61    );
62
63    while let Some(res) = task_set.join_next().await {
64        res??;
65    }
66
67    Ok(())
68}
