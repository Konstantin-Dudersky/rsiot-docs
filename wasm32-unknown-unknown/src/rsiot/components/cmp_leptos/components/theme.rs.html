1use leptos::prelude::*;
2
3const CLASS_NAMES: &str = "h-full bg-background text-on-background";
4const STYLE: &str = "color-scheme: $color_scheme";
5
6/// Переключение темы.
7///
8/// У элемента `body` необходимо задать `id=body`
9#[component]
10pub fn Theme(
11    /// Строковое именование темы
12    ///
13    /// Для Material Theme допустимые значения:
14    /// - dark-high-contrast
15    /// - dark-medium-contrast
16    /// - dark
17    /// - light-high-contrast
18    /// - light-medium-contrast
19    /// - light
20    #[prop(default = Signal::derive(|| "light".into()))]
21    theme: Signal<String>,
22
23    /// Свойство CSS color-scheme
24    html_color_scheme: Signal<String>,
25) -> impl IntoView {
26    Effect::new(move |_| {
27        let body_element = document().get_element_by_id("body").unwrap();
28
29        let theme = theme.get();
30        let class_name = format!("{} {}", CLASS_NAMES, theme);
31        body_element.set_class_name(&class_name);
32
33        let style = STYLE.replace("$color_scheme", &html_color_scheme.get());
34        body_element.set_attribute("style", &style).unwrap();
35    });
36}
