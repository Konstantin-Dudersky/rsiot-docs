18pub async fn fn_process<TMsg, TBuffer>(
19    config: Config<TMsg, TBuffer>,
20    msgbus_linker: MsgBusLinker<TMsg>,
21) -> super::Result<()>
22where
23    TMsg: 'static + MsgDataBound,
24    TBuffer: 'static + BufferBound,
25{
26    let mut task_set: JoinSet<Result<(), Error>> = JoinSet::new();
27
28    // Общие задачи обмена по шине CAN
29    let (ch_rx_send_to_can, ch_tx_recv_from_can) = CanGeneralTasks {
30        msgbus_linker,
31        buffer_default: config.buffer_default,
32        task_set: &mut task_set,
33        fn_input: config.fn_input,
34        period: config.period,
35        fn_periodic: config.fn_periodic,
36        fn_output: config.fn_output,
37        error_task_end_input: || Error::TaskEndInput,
38        error_task_end_output: || Error::TaskEndOutput,
39        error_tokio_mpsc_send: || Error::TokioSyncMpscSend,
40    }
41    .spawn();
42
43    let (ch_tx, ch_rx) = broadcast::channel::<CanFrame>(100000);
44
45    let task = shared_tasks::mpsc_to_broadcast::Task {
46        input: ch_rx_send_to_can,
47        output: ch_tx,
48    };
49    join_set_spawn(
50        &mut task_set,
51        "cmp_linux_can | mpsc_to_broadcast",
52        task.spawn().map_err(Error::TaskMpscToBroadcast),
53    );
54
55    let task = TaskSetupSendRecv {
56        input: ch_rx,
57        output: ch_tx_recv_from_can,
58        ifname: config.ifname.clone(),
59        can_settings: config.can_settings,
60        filters: config.filters,
61        async_version: config.async_version,
62    };
63    join_set_spawn(
64        &mut task_set,
65        "cmp_linux_can | setup_send_recv",
66        task.spawn(),
67    );
68
69    // let task = InterfaceInfo {
70    //     ifname: config.ifname,
71    //     period: Duration::from_millis(1000),
72    // };
73    // join_set_spawn(
74    //     &mut task_set,
75    //     "cmp_linux_can | interface_info",
76    //     task.spawn(),
77    // );
78
79    while let Some(res) = task_set.join_next().await {
80        res??;
81    }
82
83    Err(Error::TaskEnd)
